<!--/shortDescription-->

<!--paramName1-->data<!--/paramName1-->
<!--paramType1-->any<!--/paramType1-->
<!--paramDescription1-->
The row's data object.
<!--/paramDescription1-->

<!--returnType-->Boolean<!--/returnType-->
<!--returnDescription-->
**true** if the row is selected; otherwise **false**.
