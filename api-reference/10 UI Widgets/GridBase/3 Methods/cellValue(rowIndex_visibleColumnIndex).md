<!--/shortDescription-->

<!--paramType1-->Number<!--/paramType1-->
<!--paramDescription1-->

<!--/paramDescription1-->

<!--paramName2-->visibleColumnIndex<!--/paramName2-->
<!--paramType2-->Number<!--/paramType2-->
<!--paramDescription2-->
The visible index of the column to which the cell belongs.
<!--/paramDescription2-->

<!--returnType-->any<!--/returnType-->
<!--returnDescription-->
The cell's value.
